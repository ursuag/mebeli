select spis0.date_spis as date_akt, s.name as sklad_name, pg.name as grupa_name, pl.name as tovar_name, 'רע.' as ed_izm, spisl.kol_vo as kol_vo, s.id as id_sklad, pl.id as id_tovar
from SPISANIE_0 spis0, SPISANIE_LISTY spisl, sklad s, pilomat_grupa as pg, pilomat_listy as pl
where (spis0.id=spisl.id_parent) and (spis0.id_sklad=s.id) and (spisl.id_pilomat_listy=pl.id) and (pg.id=pl.id_grupa)
order by 1,2,3,4
